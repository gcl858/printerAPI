使用範例

JavaScript 程式碼 
// 方法1: 使用 Promise
const success = await startPrinting({
    systemName: '庫存管理系統',
    displayMessage: '準備列印出貨單據',
    apiKey: 'your-api-key',
    printContent: '列印內容...'
});

console.log('列印結果:', success);

// 方法2: 使用類別
const printer = new PrinterAPI();
printer.start({
    systemName: '庫存管理系統',
    displayMessage: '準備列印出貨單據',
    apiKey: 'your-api-key',
    printContent: '列印內容...',
    onComplete: (success) => {
        console.log('列印結果:', success);
    }
});
